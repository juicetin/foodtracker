Nutrition pack schema from RESEARCH.md:
```sql
CREATE TABLE foods (fdc_id INTEGER PRIMARY KEY, description TEXT NOT NULL, food_category TEXT, data_type TEXT, publication_date TEXT);
CREATE TABLE nutrients (id INTEGER PRIMARY KEY, name TEXT NOT NULL, unit TEXT NOT NULL, nutrient_nbr TEXT);
CREATE TABLE food_nutrients (food_id INTEGER REFERENCES foods(fdc_id), nutrient_id INTEGER REFERENCES nutrients(id), amount REAL NOT NULL, PRIMARY KEY (food_id, nutrient_id));
CREATE TABLE food_portions (id INTEGER PRIMARY KEY, food_id INTEGER REFERENCES foods(fdc_id), portion_description TEXT, gram_weight REAL, modifier TEXT);
CREATE VIRTUAL TABLE foods_fts USING fts5(description, food_category, content=foods, content_rowid=fdc_id);
```

       ```typescript
       export interface PackManifest {
         version: number;
         lastUpdated: string;        // ISO 8601
         packs: PackEntry[];
       }

       export interface PackEntry {
         id: string;                 // e.g., 'usda-core', 'afcd', 'yolo-v1'
         name: string;               // Human-readable
         type: 'nutrition' | 'model';
         version: string;            // Semver
         sizeBytes: number;          // Download size (compressed)
         sha256: string;             // Integrity hash
         url: string;                // R2 relative path
         region?: string;            // 'AU', 'UK', 'FR'
         locale?: string;            // 'en-AU', 'en-GB', 'fr'
         description: string;
         requiredAppVersion?: string;
       }

       export interface InstalledPack {
         id: string;
         name: string;
         type: 'nutrition' | 'model';
         version: string;
         filePath: string;
         sizeBytes: number | null;
         sha256: string | null;
         region: string | null;
         installedAt: string;
         lastChecked: string | null;
       }

       export interface DownloadProgress {
         totalBytesWritten: number;
         totalBytesExpected: number;
         fraction: number;
       }

       export type PackStatus = 'not-installed' | 'downloading' | 'installed' | 'update-available' | 'error';
       ```

       - `downloadPack(pack: PackEntry, onProgress: (p: DownloadProgress) => void): Promise<InstalledPack>`:

         - After download: verify SHA-256 hash matches `pack.sha256` (use expo-crypto or expo-file-system digest)

       - `getInstalledPack(packId: string): Promise<InstalledPack | null>` -- single pack lookup
       - `isPackInstalled(packId: string): Promise<boolean>`
       - `deletePack(packId: string): Promise<void>` -- delete file + remove from installedPacks table
       - `getPackFilePath(packId: string): Promise<string | null>` -- return local file path for installed pack
