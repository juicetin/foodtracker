```typescript
// Bounding box in normalized coordinates (0-1)
export interface BoundingBox {
  x: number; y: number; w: number; h: number;
}

// Confidence levels per locked decision: green >=80%, yellow 50-79%, red <50%
export type ConfidenceLevel = 'high' | 'medium' | 'low';

// Portion estimation result (mirrors Python PortionEstimate)
export interface PortionEstimate {
  weightG: number;
  confidence: string; // 'high' | 'medium' | 'low'
  method: string; // 'geometry' | 'user_history' | 'usda_default'
  suggestReference: boolean;
  details: Record<string, unknown>;
}

// Single detected food item
export interface DetectedItem {
  id: string;
  className: string;
  confidence: number; // 0-1
  bbox: BoundingBox;
  portionEstimate: PortionEstimate;
  portionMultiplier: number; // 0.5-3.0 slider value
  isRemoved: boolean; // soft-delete for undo
  removedAt?: number; // timestamp for undo timeout
  correctedFrom?: string; // original class name if user corrected
}

// Inference pipeline result
export interface InferenceResult {
  items: DetectedItem[];
  inferenceTimeMs: number;
  pipelineStages: PipelineStage[];
}

export interface PipelineStage {
  stage: 'binary' | 'detect' | 'classify';
  timeMs: number;
}

// Raw YOLO detection before enrichment
export interface RawDetection {
  x: number; y: number; w: number; h: number;
  confidence: number;
  classId: number;
  className: string;
}

// Model set for three-stage pipeline
export interface ModelSet {
  binary: TFLiteModel;
  detect: TFLiteModel;
  classify: TFLiteModel;
}

// Opaque model handle from react-native-fast-tflite
export interface TFLiteModel {
  run: (input: ArrayBufferLike[]) => Promise<ArrayBufferLike[]>;
  runSync: (input: ArrayBufferLike[]) => ArrayBufferLike[];
}

// Meal type auto-detection per locked decision
export type MealType = 'breakfast' | 'lunch' | 'snack' | 'dinner';

// Correction history record
export interface CorrectionRecord {
  id: string;
  originalClassName: string;
  correctedClassName: string;
  confidence: number;
  correctedAt: string; // ISO 8601
}

// Detection session state shape
export interface DetectionSessionState {
  photoUri: string | null;
  photoWidth: number;
  photoHeight: number;
  items: DetectedItem[];
  isDetecting: boolean;
  mealType: MealType;
  selectedItemId: string | null;
}
```

```typescript
export function getConfidenceLevel(confidence: number): ConfidenceLevel {
  if (confidence >= 0.80) return 'high';
  if (confidence >= 0.50) return 'medium';
  return 'low';
}

export const CONFIDENCE_COLORS: Record<ConfidenceLevel, string> = {
  high: '#22C55E',   // green-500
  medium: '#EAB308', // yellow-500
  low: '#EF4444',    // red-500
};

export function autoDetectMealType(): MealType {
  const hour = new Date().getHours();
  if (hour < 10) return 'breakfast';
  if (hour < 14) return 'lunch';
  if (hour < 17) return 'snack';
  if (hour < 21) return 'dinner';
  return 'snack';
}
```

```python
#!/usr/bin/env python3
"""
Export trained YOLO models to TFLite format for mobile deployment.

Exports all three pipeline stages (binary, detect, classify) from their
training run best.pt checkpoints to FP16 TFLite models.

Per RESEARCH.md:
- Export TFLite only (not CoreML) -- react-native-fast-tflite's CoreML
  delegate handles iOS hardware acceleration from .tflite files
- Use FP16 (half=True) for good size/accuracy tradeoff
- Do NOT use nms=True (do NMS in JavaScript for portability)
- Do NOT use end2end=True (incompatible with TFLite nms)

Usage:
    python training/export_mobile.py
    python training/export_mobile.py --binary-weights runs/binary/best.pt
    python training/export_mobile.py --detect-weights runs/detect/best.pt
    python training/export_mobile.py --classify-weights runs/classify/best.pt
    python training/export_mobile.py --output-dir exports/
    python training/export_mobile.py --validate  # Run validation on exported models
"""

import argparse
import json
import logging
import os
import sys
from pathlib import Path

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
log = logging.getLogger(__name__)

TRAINING_DIR = Path(__file__).resolve().parent
RUNS_DIR = TRAINING_DIR / "runs"
DEFAULT_OUTPUT_DIR = TRAINING_DIR / "exports"
```

The script should:
1. Accept --binary-weights, --detect-weights, --classify-weights for custom paths, defaulting to runs/{stage}/best.pt
2. Accept --output-dir for export destination (default: training/exports/)
3. Accept --validate flag to run a quick forward pass on exported models
4. For each model:
   - Load via `YOLO(weights_path)` with fallback chain (yolo26n -> yolo11n -> yolov8n)
   - Export with `model.export(format='tflite', imgsz=N, half=True, nms=False)`
   - Binary/classify use imgsz=224, detect uses imgsz=640
   - Log export success with file size
5. Generate a `model_manifest.json` in output dir with metadata for each exported model (id, version, file path, size, input dimensions, num_classes)
6. If --validate, load each exported model and run a dummy forward pass to verify the exported model loads correctly
7. Handle missing weights gracefully (skip with warning, don't fail entire script)

Include docstrings explaining the anti-patterns from RESEARCH.md (why no nms=True, why FP16 not INT8, why TFLite not CoreML).
