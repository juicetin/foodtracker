export interface BoundingBox { x: number; y: number; w: number; h: number; }

export interface RawDetection {
  x: number; y: number; w: number; h: number;
  confidence: number; classId: number; className: string;
}
export interface DetectedItem {
  id: string; className: string; confidence: number;
  bbox: BoundingBox; portionEstimate: PortionEstimate;
  portionMultiplier: number; isRemoved: boolean;
  removedAt?: number; correctedFrom?: string;
}
export interface InferenceResult {
  items: DetectedItem[]; inferenceTimeMs: number;
  pipelineStages: PipelineStage[];
}
export interface PipelineStage { stage: 'binary' | 'detect' | 'classify'; timeMs: number; }
export interface TFLiteModel {
  run: (input: ArrayBufferLike[]) => Promise<ArrayBufferLike[]>;
  runSync: (input: ArrayBufferLike[]) => ArrayBufferLike[];
}
export interface ModelSet { binary: TFLiteModel; detect: TFLiteModel; classify: TFLiteModel; }
export function getConfidenceLevel(c: number): ConfidenceLevel;
export function autoDetectMealType(): MealType;
